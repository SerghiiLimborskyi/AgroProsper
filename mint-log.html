function loadMintLog(filters = {}) { const url = new URL('https://your-domain.com/api/mint-log'); if (filters.wallet) url.searchParams.append('wallet', filters.wallet); if (filters.game) url.searchParams.append('game', filters.game); if (filters.date) url.searchParams.append('date', filters.date); fetch(url) .then(res => res.json()) .then(data => { const log = document.getElementById('mint-log'); if (!Array.isArray(data.items) || data.items.length === 0) { log.textContent = '😔 Мінтів за цим фільтром не знайдено.'; return; } log.innerHTML = data.items.map(item => `

🧑‍🌾 ${item.wallet}

🎮 ${item.game}

🏆 ${item.reward}

🕒 ${new Date(item.timestamp).toLocaleString()}

`).join(''); }) .catch(() => { document.getElementById('mint-log').textContent = '❌ Помилка при завантаженні.'; }); } document.getElementById('filter-btn').onclick = () => { const wallet = document.getElementById('wallet').value.trim(); const game = document.getElementById('game').value.trim(); const date = document.getElementById('date').value; loadMintLog({ wallet, game, date }); }; loadMintLog(); // стартове завантаження
